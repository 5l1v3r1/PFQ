158 static inline
159 unsigned int pfq_symmetric_hash(const char *buf)
160 {
161  const char *ptr = buf;
162 
163  struct ethhdr const *eh = (struct ethhdr const *)(ptr);
164  if (eh->h_proto != htons(0x800))
165  return 0;
166 
167  ptr += sizeof(struct ethhdr);
168 
169  struct iphdr const *ih = (struct iphdr const *)(ptr);
170  if (ih->protocol != IPPROTO_TCP &&
171  ih->protocol != IPPROTO_UDP)
172  return (ih->saddr ^ ih->daddr);
173 
174  ptr += sizeof(ih->ihl << 2);
175 
176  struct udphdr const *uh = (struct udphdr const *)(ptr);
177  return (ih->saddr ^ ih->daddr ^ uh->source ^ uh->dest);
178 
179 }
