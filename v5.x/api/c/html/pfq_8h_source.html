160 static inline
161 unsigned int pfq_symmetric_hash(const char *buf)
162 {
163  const char *ptr = buf;
164 
165  struct ethhdr const *eh = (struct ethhdr const *)(ptr);
166  if (eh->h_proto != htons(0x800))
167  return 0;
168 
169  ptr += sizeof(struct ethhdr);
170 
171  struct iphdr const *ih = (struct iphdr const *)(ptr);
172  if (ih->protocol != IPPROTO_TCP &&
173  ih->protocol != IPPROTO_UDP)
174  return (ih->saddr ^ ih->daddr);
175 
176  ptr += sizeof(ih->ihl << 2);
177 
178  struct udphdr const *uh = (struct udphdr const *)(ptr);
179  return (ih->saddr ^ ih->daddr ^ uh->source ^ uh->dest);
180 
181 }

187 unsigned int pfq_fold(unsigned int hash, unsigned int n)
188 {
189  hash = hash ^ (hash >> 8) ^ (hash >> 16) ^ (hash >> 24);
190  return hash % n;

197  * }
198  * case 4: return hash & 3;
199  * }
