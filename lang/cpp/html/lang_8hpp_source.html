261  out << "ArgNull";

264  out << "ArgData (" << arg.ptr->forall_show() << "," << arg.size << ")";

267  out << "ArgVector[" << arg.nelem << "] (" << arg.ptr->forall_show() << "," << arg.size << ")";

270  out << "ArgString '" << arg.ptr->forall_show() << "'";

272  else {
273  out << "ArgFun " << arg.size;
274  }
275 
276  return out.str();
