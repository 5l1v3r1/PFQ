410  {
411  if (fd_ != -1)
412  throw pfq_error("PFQ: socket already open");

414  fd_ = ::socket(PF_Q, SOCK_RAW, htons(ETH_P_ALL));

416  throw pfq_error("PFQ: module not loaded");

418  // allocate pdata
419 
420  data_.reset(new pfq_data { -1,
421  -1,
422  nullptr,

425  0,

441  throw pfq_error(errno, "PFQ: get id error");

443  // set Rx queue slots
444 
445  if (::setsockopt(fd_, PF_Q, Q_SO_SET_RX_SLOTS, &rx_slots, sizeof(rx_slots)) == -1)

448  data_->rx_slots = rx_slots;

450  // set caplen
451  //

454  throw pfq_error(errno, "PFQ: set Rx caplen error");

501  void
502  enable()
503  {
504  size_t tot_mem; socklen_t size = sizeof(tot_mem);
505 
506  if (data()->shm_addr != 0 &&
507  data()->shm_addr != MAP_FAILED)
508  throw pfq_error(errno, "PFQ: queue already enabled");
509 
510  if (::getsockopt(fd_, PF_Q, Q_SO_GET_SHMEM_SIZE, &tot_mem, &size) == -1)
511  throw pfq_error(errno, "PFQ: queue memory error");
512 
513  hd_ = ::open(("/dev/hugepages/pfq." + std::to_string(fd_)).c_str(), O_CREAT | O_RDWR, 0755);
514  if (hd_ != -1)
515  data()->shm_addr = ::mmap(nullptr, tot_mem, PROT_READ|PROT_WRITE, MAP_SHARED, hd_, 0);
516 
517  if (data()->shm_addr == MAP_FAILED)
518  {
519  void * null = nullptr;
520  if(::setsockopt(fd_, PF_Q, Q_SO_ENABLE, &null, sizeof(null)) == -1) {
521  throw pfq_error(errno, "PFQ: socket enable");
522  }
523 
524  data()->shm_addr = ::mmap(nullptr, tot_mem, PROT_READ|PROT_WRITE, MAP_SHARED, fd_, 0);
525  }
526  else
527  {
528  if(::setsockopt(fd_, PF_Q, Q_SO_ENABLE, &data()->shm_addr, sizeof(data()->shm_addr)) == -1) {
529  throw pfq_error(errno, "PFQ: socket enable");
530  }
531  }
532 
533  if (data()->shm_addr == MAP_FAILED)
534  throw pfq_error(errno, "PFQ: socket enable (mmap)");
535 
536  data()->shm_size = tot_mem;
537 
538  data()->rx_queue_addr = static_cast<char *>(data()->shm_addr) + sizeof(pfq_queue_hdr);
539  data()->rx_queue_size = data()->rx_slots * data()->rx_slot_size;
540 
541  data()->tx_queue_addr = static_cast<char *>(data()->shm_addr) + sizeof(pfq_queue_hdr) + data()->rx_queue_size * 2;
542  data()->tx_queue_size = data()->tx_slots * data()->tx_slot_size;
543  }
